# BuildinStateInterpreter

## Declaration

```typescript
export class BuildinStateInterpreter extends CommandInterpreter {
  constructor(state: TableState)
}
```

## Purpose

Use `BuildinStateInterpreter` to execute table commands directly on a `TableState` instance. This is the built-in implementation of `CommandInterpreter` for in-memory table operations.

**When to use:**
- When executing commands generated by `TableCommandPlanner`
- When applying saved command sequences to a table
- When implementing undo/redo functionality
- When you need command execution on TableState

**When NOT to use:**
- Do NOT use when you need custom storage (extend `CommandInterpreter` instead)
- Do NOT create multiple instances for the same table (share the instance)

## Constructor

### `constructor(state: TableState)`

Creates a new interpreter linked to the specified `TableState`.

**Parameters:**
- `state: TableState` - The table state to operate on

**Behavior:**
- Stores reference to the `TableState`
- All commands modify this `TableState` directly
- Changes are immediately reflected in the `TableState`

**Example:**
```typescript
const table = new TableState(10, 10)
const interpreter = new BuildinStateInterpreter(table)
```

## Implementation Details

`BuildinStateInterpreter` provides concrete implementations of all abstract `CommandInterpreter` methods:

| Method | Implementation |
|--------|----------------|
| `handleInsertRow` | Inserts rows and adjusts merged cells |
| `handleInsertCol` | Inserts columns and adjusts merged cells |
| `handleDeleteRow` | Deletes rows and adjusts merged cells |
| `handleDeleteCol` | Deletes columns and adjusts merged cells |
| `handleSetCellAttr` | Sets cell attributes via `TableState.updateCell()` |
| `handleClearCellAttr` | Clears cell attributes via `TableState.updateCell()` |

**Key features:**
- Automatically adjusts merged cells during structural operations
- Maintains table consistency
- Uses sparse storage efficiently

## Best Practices

1. **Share with TableCommandPlanner**
   - Both should reference the same `TableState` instance
   - Ensures planner and interpreter work on the same data

2. **Use for command execution only**
   - Don't use for direct cell manipulation (use `TableState` methods instead)
   - Interpreter is for executing commands, not direct state access

3. **Batch commands when possible**
   - Use `applyCommands()` instead of individual `applyCommand()` calls
   - More efficient for multiple operations

4. **Consider using TableCommandPlanner**
   - For high-level operations, use `TableCommandPlanner` to generate commands
   - Then execute generated commands with `BuildinStateInterpreter`

## Common Pitfalls

1. **Creating multiple interpreters for the same table**
   ```typescript
   // BAD: Creates separate interpreters
   const interpreter1 = new BuildinStateInterpreter(table)
   const interpreter2 = new BuildinStateInterpreter(table)
   // Both work on the same table, but this is unnecessary

   // GOOD: Share the instance
   const interpreter = new BuildinStateInterpreter(table)
   ```

2. **Not using TableCommandPlanner for complex operations**
   ```typescript
   // BAD: Manually creating merge commands
   const commands: TableCommand[] = [
     { type: 'SET_CELL_ATTR', row: 0, col: 0, attr: 'rowSpan', value: 2 },
     { type: 'SET_CELL_ATTR', row: 0, col: 0, attr: 'colSpan', value: 2 },
     // ... need to handle all covered cells
   }

   // GOOD: Use TableCommandPlanner
   const planner = new TableCommandPlanner(table)
   const commands = planner.merge(0, 0, 1, 1)
   if (commands) interpreter.applyCommands(commands)
   ```

3. **Expecting return values from applyCommands**
   ```typescript
   // BAD: applyCommands returns void
   const result = interpreter.applyCommands(commands)

   // GOOD: Check state after execution
   interpreter.applyCommands(commands)
   const newState = table.getGridData()
   ```

## Complete Example

```typescript
import { TableState, TableCommandPlanner, BuildinStateInterpreter, TableCommand } from '@aptx/table-commands-generator'

// Scenario 1: Basic setup and command execution
const table = new TableState(10, 10)
const planner = new TableCommandPlanner(table)
const interpreter = new BuildinStateInterpreter(table)

// Generate and execute merge commands
const mergeCmds = planner.merge(0, 0, 1, 2)
if (mergeCmds) {
  interpreter.applyCommands(mergeCmds)
  console.log('Merge completed')
}

// Scenario 2: Command replay (undo/redo)
class CommandHistory {
  private undoStack: TableCommand[][] = []
  private redoStack: TableCommand[][] = []

  execute(commands: TableCommand[], interpreter: BuildinStateInterpreter): void {
    // Save for undo
    this.undoStack.push([...commands])
    this.redoStack = [] // Clear redo stack

    // Execute
    interpreter.applyCommands(commands)
  }

  undo(interpreter: BuildinStateInterpreter): boolean {
    const commands = this.undoStack.pop()
    if (!commands) return false

    // Generate inverse commands (simplified)
    const inverse = this.generateInverse(commands)

    // Save for redo
    this.redoStack.push(commands)

    // Apply inverse
    interpreter.applyCommands(inverse)
    return true
  }

  redo(interpreter: BuildinStateInterpreter): boolean {
    const commands = this.redoStack.pop()
    if (!commands) return false

    // Save for undo
    this.undoStack.push(commands)

    // Execute
    interpreter.applyCommands(commands)
    return true
  }

  private generateInverse(commands: TableCommand[]): TableCommand[] {
    // Simplified inverse generation
    // Real implementation would need to track previous values
    const inverse: TableCommand[] = []

    for (let i = commands.length - 1; i >= 0; i--) {
      const cmd = commands[i]

      switch (cmd.type) {
        case 'INSERT_ROW':
          inverse.push({ type: 'DELETE_ROW', index: cmd.index, count: cmd.count })
          break
        case 'DELETE_ROW':
          inverse.push({ type: 'INSERT_ROW', index: cmd.index, count: cmd.count })
          break
        case 'INSERT_COL':
          inverse.push({ type: 'DELETE_COL', index: cmd.index, count: cmd.count })
          break
        case 'DELETE_COL':
          inverse.push({ type: 'INSERT_COL', index: cmd.index, count: cmd.count })
          break
        case 'SET_CELL_ATTR':
          inverse.push({ type: 'CLEAR_CELL_ATTR', row: cmd.row, col: cmd.col, attr: cmd.attr })
          break
        case 'CLEAR_CELL_ATTR':
          // Would need previous value - simplified
          break
      }
    }

    return inverse
  }
}

// Usage
const history = new CommandHistory()

const insertCmds = planner.insertRow(5, 2)
if (insertCmds) {
  history.execute(insertCmds, interpreter)
  console.log('Inserted 2 rows at index 5')
}

// Undo
if (history.undo(interpreter)) {
  console.log('Undone - rows removed')
}

// Redo
if (history.redo(interpreter)) {
  console.log('Redone - rows reinserted')
}

// Scenario 3: Command validation before execution
function safeExecute(
  commands: TableCommand[],
  interpreter: BuildinStateInterpreter,
  table: TableState
): boolean {
  // Validate each command
  for (const cmd of commands) {
    if (!isCommandValid(cmd, table)) {
      console.error('Invalid command:', cmd)
      return false
    }
  }

  // All commands valid, execute
  interpreter.applyCommands(commands)
  return true
}

function isCommandValid(cmd: TableCommand, table: TableState): boolean {
  switch (cmd.type) {
    case 'INSERT_ROW':
    case 'DELETE_ROW':
      return cmd.index >= 0 && cmd.index < table.getRowCount() && cmd.count >= 1
    case 'INSERT_COL':
    case 'DELETE_COL':
      return cmd.index >= 0 && cmd.index < table.getColCount() && cmd.count >= 1
    case 'SET_CELL_ATTR':
    case 'CLEAR_CELL_ATTR':
      return (
        cmd.row >= 0 &&
        cmd.row < table.getRowCount() &&
        cmd.col >= 0 &&
        cmd.col < table.getColCount()
      )
  }
}

// Usage
const testCmds: TableCommand[] = [
  { type: 'INSERT_ROW', index: 5, count: 1 },
]

if (safeExecute(testCmds, interpreter, table)) {
  console.log('Commands executed successfully')
}
```
